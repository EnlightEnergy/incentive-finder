from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from ..database import SessionLocal
from ..models import Program

router = APIRouter(tags=["public"])

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

@router.get("/programs")
def search_programs(q: str | None = None, state: str | None = None, utility: str | None = None, measure: str | None = None, limit: int = 100, db: Session = Depends(get_db)):
    qry = db.query(Program)
    if q:
        qry = qry.filter(Program.name.ilike(f"%{q}%"))
    if measure:
        qry = qry.filter(Program.tech_tags.contains([measure]))
    return [
        {
            "id": p.id,
            "name": p.name,
            "owner": p.owner,
            "url": p.url,
            "incentive_type": p.incentive_type,
            "status": p.status,
            "tech_tags": p.tech_tags,
            "sector_tags": p.sector_tags,
        }
        for p in qry.limit(limit)
    ]
EOF

cat > backend/app/routers/admin.py << 'EOF'
from fastapi import APIRouter, Depends, Header, HTTPException
from sqlalchemy.orm import Session
from ..database import SessionLocal
from ..models import Program
from ..config import settings

router = APIRouter(tags=["admin"])

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def require_api_key(x_api_key: str | None = Header(None)):
    if x_api_key != settings.API_ADMIN_KEY:
        raise HTTPException(status_code=401, detail="Invalid API key")

@router.post("/programs/publish", dependencies=[Depends(require_api_key)])
def publish_program(program_id: int, db: Session = Depends(get_db)):
    p = db.get(Program, program_id)
    if not p:
        raise HTTPException(404, "Not found")
    p.status = "open"
    db.commit()
    return {"ok": True}
EOF

from alembic import context
from sqlalchemy import engine_from_config, pool
from app.models import Base
from app.config import settings

config = context.config
config.set_main_option("sqlalchemy.url", settings.DATABASE_URL)

def run_migrations_offline():
    context.configure(target_metadata=Base.metadata, literal_binds=True)
    with context.begin_transaction():
        context.run_migrations()

def run_migrations_online():
    connectable = engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(connection=connection, target_metadata=Base.metadata)
        with context.begin_transaction():
            context.run_migrations()

if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
EOF

import { useEffect, useState } from 'react';
import axios from 'axios';
import ProgramCard from './ProgramCard';

type Props = { params:any };
export default function ResultsPanel({params}:Props){
  const [items, setItems] = useState<any[]>([]);
  useEffect(()=>{
    if(!params) return;
    const url = new URL("/api/programs", window.location.origin.replace(":5173", ":8000"));
    axios.get(url.toString(), { params: { q: params.naics, measure: params.measure }})
      .then(resp=> setItems(resp.data))
      .catch(()=> setItems([]));
  },[params]);

  if(!params) return null;
  return (
    <div className="grid">
      {items.map(p=> <ProgramCard key={p.id} program={p} />)}
      {items.length===0 && <div className="card">No programs found. Try another measure or keyword.</div>}
    </div>
  );
}
